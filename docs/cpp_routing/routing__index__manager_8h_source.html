26 // Manager for any NodeIndex <-> variable index conversion. The routing solver
27 // uses variable indices internally and through its API. These variable indices
28 // are tricky to manage directly because one Node can correspond to a multitude
29 // of variables, depending on the number of times they appear in the model, and
30 // if they're used as start and/or end points. This class aims to simplify
31 // variable index usage, allowing users to use NodeIndex instead.
32 //
33 // Usage:
34 // auto starts_ends = ...; // These are NodeIndex.
35 // RoutingIndexManager manager(/*nodes*/10, /*vehicles*/4, starts_ends);
36 // RoutingModel model(manager);
37 //
38 // Then, use 'manager.NodeToIndex(node)' whenever 'model' requires a variable
39 // index.
40 class RoutingIndexManager {
41  public:
42  typedef RoutingNodeIndex NodeIndex;
43  static const int64 kUnassigned;
44 
45  // Creates a NodeIndex to variable index mapping for a problem containing
46  // 'num_nodes', 'num_vehicles' and the given starts and ends for each vehicle.
47  // If used, any start/end arrays have to have exactly 'num_vehicles' elements.
48  RoutingIndexManager(int num_nodes, int num_vehicles, NodeIndex depot);
49  RoutingIndexManager(int num_nodes, int num_vehicles,
50  const std::vector<NodeIndex>& starts,
51  const std::vector<NodeIndex>& ends);
52  RoutingIndexManager(
53  int num_nodes, int num_vehicles,
54  const std::vector<std::pair<NodeIndex, NodeIndex> >& starts_ends);
55  ~RoutingIndexManager() {}
56 
57  int num_nodes() const { return num_nodes_; }
58  int num_vehicles() const { return num_vehicles_; }
59  int num_indices() const { return index_to_node_.size(); }
60  int64 GetStartIndex(int vehicle) const { return vehicle_to_start_[vehicle]; }
61  int64 GetEndIndex(int vehicle) const { return vehicle_to_end_[vehicle]; }
62  int64 NodeToIndex(NodeIndex node) const {
63  DCHECK_GE(node.value(), 0);
64  DCHECK_LT(node.value(), node_to_index_.size());
65  return node_to_index_[node];
66  }
67  std::vector<int64> NodesToIndices(const std::vector<NodeIndex>& nodes) const;
68  NodeIndex IndexToNode(int64 index) const {
69  DCHECK_GE(index, 0);
70  DCHECK_LT(index, index_to_node_.size());
71  return index_to_node_[index];
72  }
73  // TODO(user): Remove when removal of NodeIndex from RoutingModel is
74  // complete.
75  int num_unique_depots() const { return num_unique_depots_; }
76  std::vector<NodeIndex> GetIndexToNodeMap() const { return index_to_node_; }
77  gtl::ITIVector<NodeIndex, int64> GetNodeToIndexMap() const {
78  return node_to_index_;
79  }

93 };
