310 // This class represents a sorted list of disjoint, closed intervals. When an
311 // interval is inserted, all intervals that overlap it or that are even adjacent
312 // to it are merged into one. I.e. [0,14] and [15,30] will be merged to [0,30].
313 //
314 // Iterators returned by this class are invalidated by non-const operations.
315 //
316 // TODO(user): Templatize the class on the type of the bounds.
317 class SortedDisjointIntervalList {
318  public:
319  struct IntervalComparator {
320  bool operator()(const ClosedInterval& a, const ClosedInterval& b) const {
321  return a.start != b.start ? a.start < b.start : a.end < b.end;
322  }
323  };
324  typedef std::set<ClosedInterval, IntervalComparator> IntervalSet;
325  typedef IntervalSet::iterator Iterator;

334  //

345  // Adds the interval [start..end] to the list, and merges overlapping or
346  // immediately adjacent intervals ([2, 5] and [6, 7] are adjacent, but
347  // [2, 5] and [7, 8] are not).
348  //
349  // Returns an iterator to the inserted interval (possibly merged with others).
350  //
351  // If start > end, it does LOG(DFATAL) and returns end() (no interval added).
352  Iterator InsertInterval(int64 start, int64 end);
353 
354  // If value is in an interval, increase its end by one, otherwise insert the
355  // interval [value, value]. In both cases, this returns an iterator to the
356  // new/modified interval (possibly merged with others) and fills newly_covered
357  // with the new value that was just added in the union of all the intervals.
358  //
359  // If this causes an interval ending at kint64max to grow, it will die with a
360  // CHECK fail.
361  Iterator GrowRightByOne(int64 value, int64* newly_covered);
362 
363  // Adds all intervals [starts[i]..ends[i]]. Same behavior as InsertInterval()
364  // upon invalid intervals. There's a version with int64 and int32.

366  const std::vector<int64>& ends);
