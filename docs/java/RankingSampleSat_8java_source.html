38  static void rankTasks(CpModel model, IntVar[] starts, Literal[] presences, IntVar[] ranks) {
39  int numTasks = starts.length;
40 
41  // Creates precedence variables between pairs of intervals.
42  Literal[][] precedences = new Literal[numTasks][numTasks];
43  for (int i = 0; i < numTasks; ++i) {
44  for (int j = 0; j < numTasks; ++j) {
45  if (i == j) {
46  precedences[i][i] = presences[i];
47  } else {
48  IntVar prec = model.newBoolVar(String.format("%d before %d", i, j));
49  precedences[i][j] = prec;
50  // Ensure that task i precedes task j if prec is true.
51  model.addLessOrEqualWithOffset(starts[i], starts[j], 1).onlyEnforceIf(prec);
52  }
53  }
54  }
55 
56  // Create optional intervals.
57  for (int i = 0; i < numTasks - 1; ++i) {
58  for (int j = i + 1; j < numTasks; ++j) {
59  List<Literal> list = new ArrayList<>();
60  list.add(precedences[i][j]);
61  list.add(precedences[j][i]);
62  list.add(presences[i].not());
63  // Makes sure that if i is not performed, all precedences are false.
64  model.addImplication(presences[i].not(), precedences[i][j].not());
65  model.addImplication(presences[i].not(), precedences[j][i].not());
66  list.add(presences[j].not());
67  // Makes sure that if j is not performed, all precedences are false.
68  model.addImplication(presences[j].not(), precedences[i][j].not());
69  model.addImplication(presences[j].not(), precedences[j][i].not());
70  // The following boolOr will enforce that for any two intervals:
71  // i precedes j or j precedes i or at least one interval is not
72  // performed.
73  model.addBoolOr(list.toArray(new Literal[0]));
74  // For efficiency, we add a redundant constraint declaring that only one of i precedes j and
75  // j precedes i are true. This will speed up the solve because the reason of this
76  // propagation is shorter that using interval bounds is true.
77  model.addImplication(precedences[i][j], precedences[j][i].not());
78  model.addImplication(precedences[j][i], precedences[i][j].not());
79  }
80  }
81 
82  // Links precedences and ranks.
83  for (int i = 0; i < numTasks; ++i) {
84  IntVar[] vars = new IntVar[numTasks + 1];
85  int[] coefs = new int[numTasks + 1];
86  for (int j = 0; j < numTasks; ++j) {
87  vars[j] = (IntVar) precedences[j][i];
88  coefs[j] = 1;
89  }
90  vars[numTasks] = ranks[i];
91  coefs[numTasks] = -1;
92  // ranks == sum(precedences) - 1;
93  model.addEquality(LinearExpr.scalProd(vars, coefs), 1);
94  }
95  }

97  public static void main(String[] args) throws Exception {
98  CpModel model = new CpModel();
99  int horizon = 100;
100  int numTasks = 4;
101 
102  IntVar[] starts = new IntVar[numTasks];
103  IntVar[] ends = new IntVar[numTasks];
104  IntervalVar[] intervals = new IntervalVar[numTasks];
105  Literal[] presences = new Literal[numTasks];
106  IntVar[] ranks = new IntVar[numTasks];
107 
108  IntVar trueVar = model.newConstant(1);
109 
110  // Creates intervals, half of them are optional.
111  for (int t = 0; t < numTasks; ++t) {
112  starts[t] = model.newIntVar(0, horizon, "start_" + t);
113  int duration = t + 1;
114  ends[t] = model.newIntVar(0, horizon, "end_" + t);
115  if (t < numTasks / 2) {
116  intervals[t] = model.newIntervalVar(starts[t], duration, ends[t], "interval_" + t);
117  presences[t] = trueVar;
118  } else {
119  presences[t] = model.newBoolVar("presence_" + t);
120  intervals[t] = model.newOptionalIntervalVar(
121  starts[t], duration, ends[t], presences[t], "o_interval_" + t);
122  }
123 
124  // The rank will be -1 iff the task is not performed.
125  ranks[t] = model.newIntVar(-1, numTasks - 1, "rank_" + t);
126  }
127 
128  // Adds NoOverlap constraint.
129  model.addNoOverlap(intervals);
130 
131  // Adds ranking constraint.
132  rankTasks(model, starts, presences, ranks);
133 
134  // Adds a constraint on ranks (ranks[0] < ranks[1]).
135  model.addLessOrEqualWithOffset(ranks[0], ranks[1], 1);
136 
137  // Creates makespan variable.
138  IntVar makespan = model.newIntVar(0, horizon, "makespan");
139  for (int t = 0; t < numTasks; ++t) {
140  model.addLessOrEqual(ends[t], makespan).onlyEnforceIf(presences[t]);
141  }
142  // The objective function is a mix of a fixed gain per task performed, and a fixed cost for each
143  // additional day of activity.
144  // The solver will balance both cost and gain and minimize makespan * per-day-penalty - number
145  // of tasks performed * per-task-gain.
146  //
147  // On this problem, as the fixed cost is less that the duration of the last interval, the solver
148  // will not perform the last interval.
149  IntVar[] objectiveVars = new IntVar[numTasks + 1];
150  int[] objectiveCoefs = new int[numTasks + 1];
151  for (int t = 0; t < numTasks; ++t) {
152  objectiveVars[t] = (IntVar) presences[t];
153  objectiveCoefs[t] = -7;
154  }
155  objectiveVars[numTasks] = makespan;
156  objectiveCoefs[numTasks] = 2;
157  model.minimize(LinearExpr.scalProd(objectiveVars, objectiveCoefs));
158 
159  // Creates a solver and solves the model.
160  CpSolver solver = new CpSolver();
161  CpSolverStatus status = solver.solve(model);
162 
163  if (status == CpSolverStatus.OPTIMAL) {
164  System.out.println("Optimal cost: " + solver.objectiveValue());
165  System.out.println("Makespan: " + solver.value(makespan));
166  for (int t = 0; t < numTasks; ++t) {
167  if (solver.booleanValue(presences[t])) {
168  System.out.printf("Task %d starts at %d with rank %d%n", t, solver.value(starts[t]),
169  solver.value(ranks[t]));
170  } else {
171  System.out.printf(
172  "Task %d in not performed and ranked at %d%n", t, solver.value(ranks[t]));
173  }
174  }
175  } else {
176  System.out.println("Solver exited with nonoptimal status: " + status);
177  }
178  }
179 }
