44  }

66  for (int i = 0; i < numItems; ++i) {
67  vars[i] = x[i][b];

70  }
71

73  for (int i = 0; i < numItems; ++i) {
74  IntVar[] vars = new IntVar[numBins];
75  for (int b = 0; b < numBins; ++b) {

80
